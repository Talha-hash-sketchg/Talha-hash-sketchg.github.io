    // Basit tema toggle (localStorage ile saklar). EriÅŸilebilir: dÃ¼ÄŸme aria-pressed gÃ¼ncellenir.
    (function(){
      const btn = document.getElementById('themeToggle');
      if(!btn) return;

      const apply = (theme) => {
        if(theme === 'light'){
          document.documentElement.style.setProperty('--bg-dark','#f4f4f6');
          document.documentElement.style.setProperty('--panel','rgba(0,0,0,0.06)');
          document.documentElement.style.setProperty('--muted-white','#0b0b0b');
          document.documentElement.style.setProperty('--skill-bg','rgba(255,255,255,0.65)');
          document.documentElement.style.setProperty('--glass-border','rgba(0,0,0,0.06)');
          btn.setAttribute('aria-pressed','true');
        } else {
          document.documentElement.style.removeProperty('--bg-dark');
          document.documentElement.style.removeProperty('--panel');
          document.documentElement.style.removeProperty('--muted-white');
          document.documentElement.style.removeProperty('--skill-bg');
          document.documentElement.style.removeProperty('--glass-border');
          btn.setAttribute('aria-pressed','false');
        }
      };

      const saved = localStorage.getItem('talha-theme');
      if(saved) apply(saved);

      btn.addEventListener('click', () => {
        const current = btn.getAttribute('aria-pressed') === 'true' ? 'light' : null;
        const next = current ? 'dark' : 'light';
        apply(next);
        localStorage.setItem('talha-theme', next === 'light' ? 'light' : 'dark');
      });
    })();
  </script>
